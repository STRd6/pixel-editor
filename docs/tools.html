neighbors = (point) ->
  [
    Point(point.x, point.y-1)
    Point(point.x-1, point.y)
    Point(point.x+1, point.y)
    Point(point.x, point.y+1)
  ]

shapeTool = (fn, icon) ->
  start = null

  iconUrl: icon
  touch: ({position})->
    start = position

  move: ({editor, position})->
    editor.preview ->
      fn start, position, editor.draw

  release: ({position, editor}) ->
    fn start, position, editor.draw
